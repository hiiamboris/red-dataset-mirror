## File: `Demoplayer.red`

```Red
Red [
    Title:  "Simple Red VID samples player"
    Author: "RenaudG"
]

; List of available demos
samples: [
    "Simple drawing with circles"  https://raw.githubusercontent.com/nodrygo/DemosRed/master/simpleDraw/simpleDraw1.red
    "Simple line drawing"          https://raw.githubusercontent.com/nodrygo/DemosRed/master/simpleDraw/simpleDraw2.red
    "This demo player"             https://gist.github.com/RenaudG/bb9b9e3049dc8a0bd84b5e51b0d59768/raw/a871581e8efb8b72ba8beeaf734d3bb58d6d0e2b/Demoplayer.red
]

; 
select-demo: func [n] [
    source: read samples/(2 * n)
    size: 0x25 * (length? split source "^/") + 700x00
    reduce [source size]
]

menu: []
foreach [title url] samples [ append menu title ]

view [
    drop-down 600 data menu font-size 12 on-change [
        psize: face/parent/size - ed/size
        r: select-demo face/selected
        ed/text: r/1
        ed/size: r/2
        ;set [ed/text ed/size] select-demo face/selected
        b/enable?: true
        face/parent/size: psize + r/2
    ]
    b: button 80 "RUN" disabled [attempt [do ed/text]]
    return
    ed: text 700x50 255.255.255 font-size 12 font-name "Consolas" "Choose your sample above, then run it if you wish."
]
```
